select
        pg.id gradeId,
        pg.grade_name gradeName,
        gc.carton_code
from    glt_carton gc
join    glt_packing_grade pg on gc.initial_grade_id = pg.id
where   gc.id not in (select carton_id from glt_carton_cooling)
and     gc.add_date between cast(cast(:fromDate as text) || ' 07:00:00.000000' as timestamp)
          and cast(cast(:toDate as text) || ' 06:59:59.999999' as timestamp)
order by pg.id
