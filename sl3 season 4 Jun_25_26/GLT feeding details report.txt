-- GLT Feeding Details Report
select gfd.add_date addDate,
       gfd.bale_id baleId,
       g.grade_name gradeName,
       (vbc.bale_weight - (vbc.sacks_weight + vbc.extra_deduction_weight)) as gltReceivedWeight,
       (gfd.gross_weight - gfd.trolly_weight)          as netWeight
from glt_feeding_details gfd
         join glt_stock gs on gfd.bale_id=gs.bale_id
         and  gs.factory_id=:factoryId
         and  gfd.process_id=:processId
         join v_bale_classification_details vbc on gfd.bale_id = vbc.bale_id
         join grade g on vbc.grade_id = g.id
--full shipment
and gfd.add_date between
                            CAST(CAST(:fromDate  AS text) || ' 06:31:00.000000' AS timestamp)
                            and
                            CAST(CAST(:toDate  AS text) || ' 06:30:59.999999' AS timestamp)
--first shipment
    and gfd.add_date between
                             CAST(CAST(:fromDate  AS text) || ' 06:31:00.000000' AS timestamp)
                             and
                             CAST(CAST(:toDate  AS text) || ' 18:30:59.999999' AS timestamp)
--second shipment
    and gfd.add_date between
                             CAST(CAST(:fromDate  AS text) || ' 18:31:00.000000' AS timestamp)
                             and
                             CAST(CAST(:toDate  AS text) || ' 06:30:59.999999' AS timestamp)
order by gfd.add_date
