select
    pm.purchase_date pmDate,
    fa.area_id,
    fa.area_name,
    fa.zone_id ,
    fa.zone_name ,
    fa.sub_area_id ,
    fa.sub_area_name ,
    fa.village_id ,
    fa.village_name ,
    fa.farmer_reg_number ,
    fa.farmer_category ,
    fa.farmer_name ,
    fa.farmer_father_name ,
    fa.bank_name ,
    fa.bank_branch_name ,
    fa.bank_account_number ,
    fa.variety_name,
    b.bale_code ,
    ei.emp_name buyer_user_name,
    ei.office_phone_no buyer_mob_no,
    g.grade_name buyer_grade_name,
    coalesce(pd.actual_weight,0) actual_weight,
    bs.sacks_weight,
    coalesce(pd.weight,0) buying_weight,
    g.grade_rate,
    pd.amount as amount,
    coalesce(bb.bonus_amount,0) bonus,
    pd.amount + coalesce(bb.bonus_amount,0) total_amount,
    --reweight
    coalesce(bw.weight,0) actual_reweight,
    (coalesce(bw.weight,0) - coalesce(bsc.sacks_weight,0)) + coalesce(bsc.extra_weight,0) additional_weight,
    coalesce(bw.weight,0) - coalesce(bsc.sacks_weight,0) net_reweight,
    gc.grade_name classifier_Grade_Name,
    coalesce(eic.emp_name,'N/A') classification_Users_Name,
    coalesce(to_char(bw.add_date_time,'YYYY-MM-DD'),'N/A') classified_Date,
    bc.center_name
from    farmer_agreement fa
            join    season s on fa.season_id = s.id
    and     s.id=:season
            and (case when :areaId is not null then fa.area_id = :areaId else true end)
            and (case when :zoneId is not null then fa.zone_id = :zoneId else true end)
            and (case when :subAreaId is not null then fa.sub_area_id = :subAreaId else true end)
            and (case when :villageId is not null then fa.village_id = :villageId else true end)
            join    purchase_main pm on fa.id = pm.agreement_id
--and     pm.purchase_date between to_date(:fromDate,'yyyy-mm-dd') and to_date(:toDate,'yyyy-mm-dd')
            join    purchase_details pd on pm.id = pd.purchase_main_id
            join    buying_center bc on pm.buying_center_id = bc.id
            join    bale b on pd.bale_id = b.id
            left    join (select * from bale_grade where grade_provider='BUYER') bg on b.id = bg.bale_id
            left    join (select * from grade where grade_provider='BUYER' and season_id=:season) g on bg.grade_id = g.id
                    and fa.area_id=g.area_id and fa.variety_id=g.variety_id
            left    join emp_info ei on bg.added_by_user_id=ei.users_user_id
            left    join (select bale_id,sacks_weight from bale_sacks where sack_type_selector ='BUYER') bs on b.id = bs.bale_id
            left    join bale_bonus bb on pd.bale_id=bb.bale_id
            left    join (select  add_date_time,bale_id,weight from bale_weight where weight_recipient='RE_WEIGHT') bw on pd.bale_id=bw.bale_id
            left    join (select bale_id,sacks_weight,extra_weight from bale_sacks where sack_type_selector ='CLASSIFIER') bsc on bw.bale_id = bsc.bale_id
            left    join (select * from bale_grade where grade_provider='CLASSIFIER') bgc on b.id = bgc.bale_id
            left    join (select * from grade where grade_provider='CLASSIFIER' and season_id=:season) gc on bgc.grade_id = gc.id
                    and fa.area_id=gc.area_id and fa.variety_id=gc.variety_id
            left    join users uc on bgc.added_by_user_id=user_id
            left    join emp_info eic on uc.user_id = eic.users_user_id
order    by fa.area_id,fa.zone_id, fa.sub_area_id, fa.village_id
