select  fa.area_id as areaId, fa.area_name as areaName,
fa.zone_id as zoneId, fa.zone_name as zoneName,
        fa.sub_area_id as subAreaId, fa.sub_area_name as subAreaName,
        fa.farmer_reg_number as farRegNo, fa.farmer_name as farmerName, fa.farmer_father_name as farFatherName,
        fa.village_id as villageId, fa.village_name as villageName,
        tps.purchase_date as purchaseDate,
        coalesce(cast(pm.purchase_no as varchar), 'N/A') as voucherNumber,
        brr.bale_id as baleId,
        rr.reason as reason,
        count(brr.bale_id) nosofBale
        from farmer_agreement fa
        join season s on fa.season_id = s.id
            and s.id=:seasonId
	and (case when :areaId is not null then fa.area_id=:areaId else true end)
        join transit_pass tp on fa.id = tp.farmer_agreement_id
        join tp_purchase_schedule tps on tp.id = tps.tp_id
        join bale b on tp.id = b.transit_pass_id
            and cast(b.add_date_time as date) between to_date(:fromDate,'yyyy-mm-dd') and to_date(:toDate,'yyyy-mm-dd')
        join bale_reject_reason brr on b.id = brr.bale_id
        join reject_reason rr on brr.reject_reason_id = rr.id
        left join purchase_main pm on fa.id = pm.agreement_id and tp.id = pm.tp_no
group by fa.area_id, fa.area_name, fa.zone_id, fa.zone_name, fa.sub_area_id, fa.sub_area_name, fa.village_id, fa.village_name,
          fa.farmer_reg_number, fa.farmer_name, fa.farmer_father_name, tps.purchase_date, pm.purchase_no, brr.bale_id, rr.reason
order by fa.area_id, fa.zone_id, fa.sub_area_id, fa.village_name, tps.purchase_date, fa.farmer_reg_number
