
select  fa.area_id areaId,
        fa.area_name areaName,
        fa.zone_id zoneId,
        fa.zone_name zoneName,
        fa.sub_area_id subAreaId,
        fa.sub_area_name subAareaName,
        fa.farmer_reg_number farmerId,
        fa.farmer_name farmerName,
        fa.farmer_father_name fatherName,
        fa.village_id villageId,
        fa.village_name villageName,
        g.grade_name externalGrade,
        ing.internalGrade as internalGrade,
        count(ing.bale_id)nosofBale,
        coalesce(sum(pd.weight),0)purchaseWeight,
        coalesce(sum(pd.amount),0) amount,
        coalesce(sum(bb.bonus_amount),0) bonusAmount,
        coalesce(sum(pd.amount),0) + coalesce(sum(bb.bonus_amount),0) totalAmount
from    bale_grade bg
            join    (select * from grade where grade_provider='BUYER' and season_id=:season) g on bg.grade_id=g.id
    and     bg.grade_provider = 'BUYER'
            join    variety v on g.variety_id = v.id
    and     v.id = :varietyId
            join    purchase_details pd on bg.bale_id=pd.bale_id
            join    purchase_main pm on pd.purchase_main_id = pm.id
    and     pm.purchase_date between to_date(:fromDate,'yyyy-mm-dd') and to_date(:toDate,'yyyy-mm-dd')
            join    farmer_agreement fa on pm.agreement_id = fa.id and g.area_id=fa.area_id
    and fa.area_id = :areaId
    and (case when :zoneId is not null then fa.zone_id = :zoneId else true end)
    and (case when :subAreaId is not null then fa.sub_area_id = :subAreaId else true end)
            join    season s on fa.season_id = s.id
    and     s.id=:season
            left    join bale_bonus bb on pd.bale_id=bb.bale_id
            join    (select  cbg.bale_id,
                             cg.grade_name internalGrade
                     from    bale_grade cbg
                                 join (select * from grade where grade_provider='CLASSIFIER' and season_id=:season) cg on cbg.grade_id=cg.id
                         and     cbg.grade_provider = 'CLASSIFIER'
                                 join    purchase_details cpd on cbg.bale_id=cpd.bale_id
                                 join    purchase_main cpm on cpd.purchase_main_id = cpm.id
                         and     cpm.purchase_date between to_date(:fromDate,'yyyy-mm-dd') and to_date(:toDate,'yyyy-mm-dd')
                     ) ing on pd.bale_id=ing.bale_id
group by fa.area_id, fa.area_name, fa.zone_id, fa.zone_name, fa.sub_area_id, fa.sub_area_name,fa.farmer_reg_number,
         fa.farmer_name, fa.farmer_father_name, fa.village_id, fa.village_name, g.grade_name,ing.internalGrade
order by fa.area_id, fa.zone_id, fa.sub_area_id, fa.village_id, fa.farmer_reg_number,g.grade_name ,ing.internalGrade
