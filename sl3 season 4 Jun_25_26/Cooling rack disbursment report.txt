select
        toWarehouseId godownId,
        toWarehouseName godownName,
        gradeId gradeId,
        gradeName packingGrade,
        count(carton_id) cartDisQty,
        sum(netWeight) cartDisWeight
from (select
          gpg.id gradeId,
          gpg.grade_name gradeName,
          td.carton_id,
          gw.id toWarehouseId,
          gw.warehouse_name toWarehouseName,
          gc.net_weight netWeight,
          rank () over (partition by td.carton_id order by td.add_date) rank
      from    glt_carton_cooling gcc
                  join    glt_wh_transfer_details td on gcc.carton_id=td.carton_id
          and     td.add_date between cast(cast(:fromDate as text) || ' 00:00:00.000000' as timestamp)
                                                            and cast(cast(:toDate as text) || ' 23:59:59.999999' as timestamp)
                  join    glt_wh_transfer gt on td.transfer_id = gt.id
                  join    glt_warehouse gw on gt.to_warehouse_id = gw.id
                  join    glt_carton gc on td.carton_id = gc.id
                  join glt_packing_grade gpg on gc.initial_grade_id = gpg.id
      where   gcc.status='DISPATCHED'
      ) tow
where rank=1
group by toWarehouseId, toWarehouseName,gradeId, gradeName
order by toWarehouseId, gradeId
