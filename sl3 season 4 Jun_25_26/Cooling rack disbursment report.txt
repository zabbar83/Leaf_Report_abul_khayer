select
    gw.id godownName,
    gw.warehouse_name godownName,
    gpg.id gradeId,
    gpg.grade_name packingGrade,
    count(gc.id) cartDisQty,
    sum(gc.net_weight) cartDisWeight
from    glt_wh_transfer gt
            join    glt_wh_transfer_details gwtd on gt.id = gwtd.transfer_id
    and     gt.cooling_rack_id is not null
    and     gt.add_date between cast(cast(:fromDate as text) || ' 00:00:00.000000' as timestamp)
                                                        and cast(cast(:toDate as text) || ' 23:59:59.999999' as timestamp) + INTERVAL '1 day'
            join    glt_carton gc on gwtd.carton_id = gc.id
    and     gc.status not in('CANCELLED', 'RF_COMPLETED')
            join    glt_warehouse gw on gt.to_warehouse_id = gw.id
            join    glt_packing_grade gpg on gc.initial_grade_id = gpg.id
group by gw.id, gw.warehouse_name, gpg.id, gpg.grade_name
order   by gw.id,gpg.id
