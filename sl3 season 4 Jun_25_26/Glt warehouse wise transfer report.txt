select
        cast(gt.add_date as date) addDate,
        gpg.grade_name packingGrade,
        gw.warehouse_name recGodownName,
        count(gwtd.id) sendQty,
        count(case when gwtd.status='RECEIVED' then gwtd.id else null end) receivedQty,
        count(case when gwtd.status not in ('RECEIVED') then gwtd.id else null end) notReceivedQty
from    glt_wh_transfer gt
join    glt_wh_transfer_details gwtd on gt.id = gwtd.transfer_id
and     gt.add_date between cast(cast(:fromDate as text) || ' 00:00:00.000000' as timestamp)
        and cast(cast(:toDate as text) || ' 23:59:59.999999' as timestamp)
join    glt_warehouse gw on gt.to_warehouse_id = gw.id
and     gw.glt_id=:gltId
join    glt_carton gc on gwtd.carton_id = gc.id
join    glt_packing_grade gpg on gc.initial_grade_id = gpg.id
group   by cast(gt.add_date as date), gpg.grade_name,gw.warehouse_name
order   by cast(gt.add_date as date)
