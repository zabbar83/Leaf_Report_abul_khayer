select
    tb.chalanDate as chalanDate,
    fbc.center_name fromName,
    f.factory_name toName,
    tb.chalan_no challanNo,
    t.vehicle_reg_no truckNo,
    max(tb.total_bale) challanQtyBale,
    count(tb.bale_id) recvQtyBale,
    max(tb.total_bale) - count(tb.bale_id) scanePendingBale
from
    (select  cast(gs.add_date_time as date)chalanDate,
             bti.factory_id,
             bti.from_buying_center_id,
             bti.transport_id,
             bti.chalan_no,
             btit.bale_id,
             SUM(count(btit.bale_id )) OVER (PARTITION BY bti.chalan_no) total_bale
     from    bale_transfer_info bti
                 join    bale_transfer_item btit
                         on bti.id = btit.transfer_id
                             and bti.chalan_no in (select  distinct bti.chalan_no
                                                   from    bale_transfer_info bti
                                                               join    bale_transfer_item btit
                                                                       on bti.id = btit.transfer_id
                                                               join    glt_stock gs on  btit.bale_id = gs.bale_id
                                                       and     cast(gs.add_date_time as date) between to_date(:fromDate,'YYYY-MM-DD') and to_date(:toDate,'YYYY-MM-DD')
                             )
                 left join    glt_stock gs on  btit.bale_id = gs.bale_id
         and     cast(gs.add_date_time as date) between to_date(:fromDate,'YYYY-MM-DD') and to_date(:toDate,'YYYY-MM-DD')
     group by cast(gs.add_date_time as date),
              bti.factory_id,
              bti.from_buying_center_id,
              bti.transport_id,
              bti.chalan_no,
              btit.bale_id) tb
        join     factory f on tb.factory_id=f.id
        and     f.id=:factoryId
        join    buying_center fbc on tb.from_buying_center_id = fbc.id
        and     (case when :fromId is not null then tb.from_buying_center_id=:fromId else true end)
        join    transport t on tb.transport_id = t.id
group by tb.chalanDate,fbc.center_name, f.factory_name, tb.chalan_no, t.vehicle_reg_no
order by tb.chalanDate desc
