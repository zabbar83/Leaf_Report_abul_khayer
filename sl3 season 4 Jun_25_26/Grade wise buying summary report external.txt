-- external buyingCenterWise last update
select
        v.id as varietyId,
        v.variety_name as varietyName,
        g.id as gradeId,
        g.grade_name as gradeName,
        count(pd.bale_id) nosofBale,
        sum(pd.weight) as totalWeight,
        SUM(SUM(pd.weight )) OVER (PARTITION BY v.id) AS varietyTotalWeight,
        sum(pd.amount) as amount,
        coalesce(sum(bb.bonus_amount),0) as bonusAmount,
        sum(pd.amount) + coalesce(sum(bb.bonus_amount),0) totalAmount,
        (sum(pd.weight )/(SUM(SUM(pd.weight )) OVER (PARTITION BY v.id)))*100 AS contribution
from    farmer_agreement fa
join    purchase_main pm
        on fa.id = pm.agreement_id
        and fa.season_id = 4
          and (case when 166 is not  null then  pm.buying_center_id =166 else true end)
                              and (case when 3 is not null then fa.area_id = 3 else true end)
                              and (case when null is not null then fa.zone_id = null else true end)
                              and (case when null is not null then fa.sub_area_id = null else true end)
                              and (case when null is not null then fa.village_id = null else true end)
        and pm.purchase_date between to_date('2026-04-01','yyyy-mm-dd') and to_date('2026-04-01','yyyy-mm-dd')
join    purchase_details pd
        on pm.id = pd.purchase_main_id
join    grade g
        on pd.grade_id = g.id
        and g.season_id = fa.season_id
        and g.area_id = fa.area_id
        and g.grade_provider = 'BUYER'
        and g.variety_id in(1)
join    variety v
        on g.variety_id = v.id
left    join bale_bonus bb
        on pd.bale_id = bb.bale_id
group   by v.id, v.variety_name, g.id, g.grade_name
