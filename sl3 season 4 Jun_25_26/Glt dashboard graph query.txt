
select
            gltId,
            gradeId,
            gradeName,
            nosofBale,
            feddingIssueWeight,
            totalFeddingIssueWeight,
            blandPersentage
from    (
            select
                gft.glt_id as gltId,
                cbg.grade_id as gradeId,
                cbg.grade_name as gradeName,
                count(cbg.grade_id) as nosofBale,

                ROUND(CAST(sum(gfd.gross_weight - gfd.trolly_weight) AS NUMERIC), 2) as feddingIssueWeight,
                ROUND(CAST(SUM(sum(gfd.gross_weight - gfd.trolly_weight)) OVER () AS NUMERIC), 2) as totalFeddingIssueWeight,

                ROUND(CAST((sum(gfd.gross_weight - gfd.trolly_weight)
                    /
                            SUM(sum(gfd.gross_weight - gfd.trolly_weight)) OVER ()
                               ) * 100  AS NUMERIC), 2) as blandPersentage
            from    (
                        SELECT
                            fb.bale_id,
                            fb.grade_id,
                            g.grade_name,
                            fb.grade_provider,
                            fb.add_date_time,
                            fb.grade_provided_by,
                            fb.re_cl_status,
                            fb.clg_rank
                        FROM    (
                                    SELECT
                                        tb.bale_id,
                                        tb.grade_id,
                                        tb.grade_provider,
                                        tb.add_date_time,
                                        tb.grade_provided_by,
                                        tb.re_cl_status,
                                        rank() OVER (PARTITION BY tb.bale_id ORDER BY tb.add_date_time DESC) AS clg_rank
                                    FROM    (
                                                SELECT
                                                    bale_grade.bale_id,
                                                    bale_grade.grade_id,
                                                    bale_grade.grade_provider,
                                                    bale_grade.add_date_time,
                                                    bale_grade.added_by_user_id AS grade_provided_by,
                                                    0 AS re_cl_status
                                                FROM    bale_grade
                                                WHERE   bale_grade.grade_provider = 'CLASSIFIER'
                                                UNION   ALL
                                                SELECT
                                                    reclassified_bale_grade.bale_id,
                                                    reclassified_bale_grade.grade_id,
                                                    reclassified_bale_grade.grade_provider,
                                                    reclassified_bale_grade.add_date_time,
                                                    reclassified_bale_grade.added_by_user_id AS grade_provided_by,
                                                    1 AS re_cl_status
                                                FROM    reclassified_bale_grade
                                            ) as tb
                                ) as fb
                                    JOIN    grade g ON fb.grade_id = g.id
                            and fb.clg_rank = 1
                    ) as cbg
                        join    glt_feeding_details gfd
                                on cbg.bale_id = gfd.bale_id
                                    and gfd.add_date between :fromDate and :toDate
                                    and (:processId is null or gfd.process_id = :processId)
                        join    glt_feeding_table gft
                                on gfd.feeding_table_id = gft.id
            group   by gft.glt_id, cbg.grade_id, cbg.grade_name
        ) as cgd
where   gltId = :gltId
order   by cgd.gradeId
