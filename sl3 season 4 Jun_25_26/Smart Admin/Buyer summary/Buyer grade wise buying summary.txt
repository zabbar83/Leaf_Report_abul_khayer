select  g.id gradeId,g.grade_name gradeName,
        count(case when pm.purchase_date = current_date then pd.bale_id end) nosofBaleToday,
        case when
                 sum(case when pm.purchase_date = current_date then pd.weight else 0 end) > 0 then
                 sum(case when pm.purchase_date = current_date then pd.amount else 0 end) /
                 sum(case when pm.purchase_date = current_date then pd.weight else 0 end)
             else 0 end avRateToday,

        count(pd.bale_id)                                                    nosofBaleTodate,
        case when sum(pd.weight) > 0 then sum(pd.amount)/sum(pd.weight) else 0 end avRaTodate
from    farmer_agreement fa
join    purchase_main pm on fa.id=pm.agreement_id
and     fa.season_id=(select  s.id
                      from season s
                               join global_setting gs on s.id=cast(gs.attribute_value as numeric)
                      where gs.attribute_name='BUYING_SEASON')
join    purchase_details pd on pm.id = pd.purchase_main_id
join    bale_grade bg on pd.bale_id=bg.bale_id
join    (select * from grade where grade_provider='BUYER') g on bg.grade_id=g.id
group by g.id,g.grade_name
order by g.id
