select tg.area_name    as areaName,
       tg.variety_name as varietyName,
       tg.id              gradeId,
       tg.grade_name   as gradeName,
       tg.depot_id        depotId,
       tg.depotName    as depotName,
       bp.nosofBale    as nosofBale,
       bp.clWeight     as clWeight
from (select distinct g.id,
                      g.grade_name,
                      g.variety_id,
                      v.variety_name,
                      vta.area_id,
                      vta.area_name,
                      d.id         depot_id,
                      d.depot_name depotName
      from grade g
               join (select distinct area_id, area_name from v_teritory_all where area_id = :areaId) vta
                    on g.area_id = vta.area_id
               join depot d on g.area_id = d.area_id
               join variety v on g.variety_id = v.id
      where grade_provider = 'CLASSIFIER'
        and g.variety_id = :varietyId
        and g.id = :gradeId
        and season_id = (select cast(attribute_value as numeric) sId
                         from global_setting
                         where attribute_name = 'BUYING_SEASON')) tg
         join (select g.area_id,
                      g.variety_id,
                      d.id                                  as depot_id,
                      d.depot_name                          as depotName,
                      count(bw.bale_id)                        nosofBale,
                      sum(bw.weight) - sum(bs.sacks_weight) as clWeight
               from (select *
                     from grade
                     where grade_provider = 'CLASSIFIER'
                       and id = :gradeId
                       and variety_id = :varietyId
                       and area_id = :areaId) g
                        join bale_grade bg on g.id = bg.grade_id
                        join (select * from bale_weight where weight_recipient = 'PURCHASE') bw
                             on bg.bale_id = bw.bale_id
                                 and bg.grade_provider = 'CLASSIFIER'
                        join (select * from bale_sacks where sack_type_selector = 'CLASSIFIER') bs
                             on bw.bale_id = bs.bale_id
                        join buying_center bc on bg.buying_center_id = bc.id
                        join depot d on bc.depot_id = d.id
               group by g.area_id, g.variety_id, d.id, d.depot_name) bp on tg.depot_id = bp.depot_id
order by tg.id, tg.depot_id
