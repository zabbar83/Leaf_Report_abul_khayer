select
    sl,
    areaId,
    areaName,
    varietyId,
    varietyName,
    clBale,
    clQty,
    nonClBale,
    nonClQty
from    (select
             1 sl,
             vta.area_id areaId,
             vta.area_name areaName,
             v.id varietyId,
             v.variety_name varietyName,
             count(bg.bale_id) clBale,
             sum(bg.bale_weight) - sum(bg.extra_deduction_weight+bg.sacks_weight) clQty,
             count(case when bcs.received_status NOT IN ('FRESH', 'REVISE_BALE') then bg.bale_id else null end) nonClBale,
             coalesce(sum(case when bcs.received_status NOT IN ('FRESH', 'REVISE_BALE') then bg.bale_weight else null end),0) -
             coalesce(sum(case when bcs.received_status NOT IN ('FRESH', 'REVISE_BALE') then (bg.extra_deduction_weight+bg.sacks_weight) else null end),0) nonClQty
             from    v_bale_classification_details bg
                     join    (select * from buying_center_stock where in_out_status IN ('IN', 'OUT_FOR_CLASSIFY')) bcs on bg.bale_id=bcs.bale_id
                     join    (select * from grade where grade_provider='CLASSIFIER' and season_id=:seasonId) g on bg.grade_id=g.id
                     join    (select distinct area_id, area_name from v_teritory_all) vta on g.area_id=vta.area_id
                     join    variety v on g.variety_id=v.id
         group by vta.area_id, vta.area_name, v.id, v.variety_name
         UNION ALL
         select
             2 sl,
             f.id areaId,
             f.factory_name areaName,
             v.id varietyId,
             v.variety_name varietyName,
             count(fs.bale_id) clBale,
             sum(bg.bale_weight) - sum(bg.extra_deduction_weight+bg.sacks_weight) clQty,
             0 nonClBale,
             0 nonClQty
             from    factory f
                     join    (select  * from glt_stock where status='IN') fs on f.id = fs.factory_id
                     join    v_bale_classification_details bg on fs.bale_id=bg.bale_id
                     join    (select * from grade where grade_provider='CLASSIFIER' and season_id=:seasonId) g on bg.grade_id=g.id
                     join    variety v on g.variety_id=v.id
         group by f.id, f.factory_name, v.id, v.variety_name) t
order by sl,areaId,varietyId
