select
    v.id varietyId,
    v.variety_name varietyName,
    bc.id buyingCenterId,
    bc.center_name buyingCenterName,
    g.id gradeId,
    g.grade_name gradeName,
    count(bg.bale_id) clBale,
    sum(bw.weight) - sum(bs.sacks_weight) clQty,
    count(case when cbg.bale_id is null then bg.bale_id else null end) as nonClBale,
    sum(case when cbg.bale_id is null then bw.weight else 0 end)
        - sum((case when cbg.bale_id is null then nbs.sacks_weight else 0 end)+(case when cbg.bale_id is null then nbs.extra_weight else 0 end)) as nonClQty

from    bale_grade bg
            join    (select * from buying_center_stock where in_out_status IN ('IN', 'OUT_FOR_CLASSIFY')) bcs on bg.bale_id=bcs.bale_id
    and     bg.grade_provider='CLASSIFIER'
            join    buying_center bc on bg.buying_center_id = bc.id
    and     bc.id=:buyingCenterId
            join    (select * from grade where grade_provider='CLASSIFIER' and season_id=:seasonId) g on bg.grade_id=g.id
            join    (select distinct area_id, area_name from v_teritory_all where area_id=:areaId) vta on g.area_id=vta.area_id
            join    variety v on g.variety_id=v.id
            and     v.id=:varietyId
            join    (select * from bale_weight where weight_recipient='CLASSIFICATION_FIRST') bw on bg.bale_id=bw.bale_id
            join    (select * from bale_sacks where sack_type_selector='CLASSIFIER') bs on bw.bale_id=bs.bale_id
            left join    (select * from bale_grade where grade_provider != 'CLASSIFIER') cbg on bg.bale_id = cbg.bale_id
            left join    (select * from bale_weight where weight_recipient = 'CLASSIFICATION_FIRST') nbw on bg.bale_id = nbw.bale_id
            left join    (select * from bale_sacks where sack_type_selector = 'CLASSIFIER') nbs on nbw.bale_id = nbs.bale_id
group by  v.id, v.variety_name,bc.id,bc.center_name,g.id, g.grade_name
order by g.id
