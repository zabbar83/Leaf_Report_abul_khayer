select
        f.id factoryId,
        f.factory_name,
        v.id varietyId,
        v.variety_name,
        count(bw.bale_id) bale,
        sum(bw.weight) qty
from    factory f
join    factory_stock fs on f.id = fs.factory_id
join    bale_grade bg on fs.bale_id=bg.bale_id
join    (select * from grade where grade_provider='CLASSIFIER') g on bg.grade_id=g.id
join    variety v on g.variety_id=v.id
join    (select * from bale_weight where weight_recipient='RE_WEIGHT') bw on bg.bale_id=bw.bale_id
group by f.id, f.factory_name, v.id, v.variety_name
order by f.id,v.id
