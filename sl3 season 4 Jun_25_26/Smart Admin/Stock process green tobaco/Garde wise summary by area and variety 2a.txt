select tg.area_name as areaName, v.variety_name as varietyName,
       tg.id gradeId, tg.grade_name as gradeName,
       count(bw.bale_id) nosofBale,
       sum(bw.weight) - sum(bs.sacks_weight) as clWeight
from (select *
      from grade g
        join (select distinct area_id, area_name from v_teritory_all where area_id=:areaId) vta on g.area_id = vta.area_id
      where grade_provider = 'CLASSIFIER'
        and g.variety_id=:varietyId
        and season_id = (select cast(attribute_value as numeric) sId
                         from global_setting
                         where attribute_name = 'BUYING_SEASON')) tg
         join bale_grade bg on tg.id = bg.grade_id
            and bg.grade_provider = 'CLASSIFIER'
         join variety v on tg.variety_id = v.id
         join (select * from bale_weight where weight_recipient = 'PURCHASE') bw on bg.bale_id = bw.bale_id
         join (select * from bale_sacks where sack_type_selector = 'CLASSIFIER') bs on bw.bale_id = bs.bale_id
group by tg.area_name, v.variety_name, tg.id, tg.grade_name
order by tg.id
