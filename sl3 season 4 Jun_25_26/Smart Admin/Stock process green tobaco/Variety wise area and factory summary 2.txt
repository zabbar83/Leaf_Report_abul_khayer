
select
        vta.area_id areaId,
        vta.area_name areaName,
        v.id varietyId,
        v.variety_name varietyName,
        count(bg.bale_id) bale,
        sum(bw.weight) - sum(bs.sacks_weight) qty,
        count(case when cbg.bale_id is null then bg.bale_id else null end) as nosofBale,
        sum(case when cbg.bale_id is null then bw.weight else 0 end)
            - sum((case when cbg.bale_id is null then nbs.sacks_weight else 0 end)+(case when cbg.bale_id is null then nbs.extra_weight else 0 end)) as purchaseWeight

from    bale_grade bg
join    (select * from buying_center_stock where in_out_status IN ('IN', 'OUT_FOR_CLASSIFY')) bcs on bg.bale_id=bcs.bale_id
and     bg.grade_provider='CLASSIFIER'
join    (select * from grade where grade_provider='CLASSIFIER' and season_id=:seasonId) g on bg.grade_id=g.id
join    (select distinct area_id, area_name from v_teritory_all) vta on g.area_id=vta.area_id
join    variety v on g.variety_id=v.id
join    (select * from bale_weight where weight_recipient='CLASSIFICATION_FIRST') bw on bg.bale_id=bw.bale_id
join    (select * from bale_sacks where sack_type_selector='CLASSIFIER') bs on bw.bale_id=bs.bale_id
left join    (select * from bale_grade where grade_provider != 'CLASSIFIER') cbg on bg.bale_id = cbg.bale_id
left join    (select * from bale_weight where weight_recipient = 'CLASSIFICATION_FIRST') nbw on bg.bale_id = nbw.bale_id
left join    (select * from bale_sacks where sack_type_selector = 'CLASSIFIER') nbs on nbw.bale_id = nbs.bale_id
group by vta.area_id, vta.area_name, v.id, v.variety_name
UNION ALL
select
        f.id areaId,
        f.factory_name areaName,
        v.id varietyId,
        v.variety_name varietyName,
        count(fs.bale_id) clBale,
        sum(bw.weight) - sum(bs.sacks_weight) clQty,
        count(case when cbg.bale_id is null then bg.bale_id else null end) as nonBale,
        sum(case when cbg.bale_id is null then bw.weight else 0 end)
            - sum((case when cbg.bale_id is null then nbs.sacks_weight else 0 end)+(case when cbg.bale_id is null then nbs.extra_weight else 0 end)) as nonQty

from    factory f
join    factory_stock fs on f.id = fs.factory_id
join    bale_grade bg on fs.bale_id=bg.bale_id and bg.grade_provider='CLASSIFIER'
join    (select * from grade where grade_provider='CLASSIFIER' and season_id=:seasonId) g on bg.grade_id=g.id
join    variety v on g.variety_id=v.id
join    (select * from bale_weight where weight_recipient='CLASSIFICATION_FIRST') bw on bg.bale_id=bw.bale_id
join    (select * from bale_sacks where sack_type_selector='CLASSIFIER') bs on bw.bale_id=bs.bale_id
left join    (select * from bale_grade where grade_provider != 'CLASSIFIER') cbg on bg.bale_id = cbg.bale_id
left join    (select * from bale_weight where weight_recipient = 'CLASSIFICATION_FIRST') nbw on bg.bale_id = nbw.bale_id
left join    (select * from bale_sacks where sack_type_selector = 'CLASSIFIER') nbs on nbw.bale_id = nbs.bale_id
group by f.id, f.factory_name, v.id, v.variety_name
