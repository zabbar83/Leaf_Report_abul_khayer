
select  fa.area_id areaId,
        fa.area_name areaName,
        fa.zone_id zoneId,
        fa.zone_name zoneName,
        count(pd.bale_id)nosofBaleEx,
        coalesce(sum(pd.weight),0)qtyEx,
        count(ing.bale_id)nosofBaleIn,
        coalesce(sum(ing.clWeight),0)qtyIn,
        count(pd.bale_id)-count(ing.bale_id) baleVarience,
        coalesce(sum(pd.weight),0)-coalesce(sum(ing.clWeight),0) qtyVarience
from    bale_grade bg
            join    (select * from grade where grade_provider='BUYER' and season_id=(select cast(attribute_value as numeric) sId
                                                                                     from global_setting
                                                                                     where attribute_name='BUYING_SEASON')) g on bg.grade_id=g.id
    and     bg.grade_provider = 'BUYER'
            join    variety v on g.variety_id = v.id
            join    purchase_details pd on bg.bale_id=pd.bale_id
            join    purchase_main pm on pd.purchase_main_id = pm.id
            join    farmer_agreement fa on pm.agreement_id = fa.id and g.area_id=fa.area_id
    and     fa.area_id=:areaId
            join    season s on fa.season_id = s.id
            join    global_setting gs on s.id=cast(attribute_value as numeric)
    and     gs.attribute_name='BUYING_SEASON'
            left    join bale_bonus bb on pd.bale_id=bb.bale_id
            join    (select  cbg.bale_id,
                             bw.weight,
                             bs.sacks_weight,
                             bw.weight - bs.sacks_weight clWeight
                     from    bale_grade cbg
                                 join (select * from grade where grade_provider='CLASSIFIER' and season_id=(select cast(attribute_value as numeric) sId
                                                                                                            from global_setting
                                                                                                            where attribute_name='BUYING_SEASON')) cg on cbg.grade_id=cg.id
                         and     cbg.grade_provider = 'CLASSIFIER'
                                 join   (select * from bale_weight where weight_recipient='RE_WEIGHT') bw on cbg.bale_id=bw.bale_id
                                 left   join (select * from bale_sacks where sack_type_selector='CLASSIFIER') bs on bw.bale_id=bs.bale_id
) ing on pd.bale_id=ing.bale_id
group by fa.area_id, fa.area_name,fa.zone_id,fa.zone_name
order by fa.area_id,fa.zone_id
