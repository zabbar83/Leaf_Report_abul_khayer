select
        bt.chalan_date chalanDate,
        bt.chalan_no shipmentNumber,
        tv.vendor_id vendorId,
        tv.trade_license_name vendorName,
        t.id transportId,
        coalesce(t.vehicle_name, 'N/A') vehicleName,
        t.vehicle_reg_no truckNumber,
       bt.from_buying_center_id shipmentFromId,
       bt.center_name shipmentFromName,
       coalesce(bt.to_buying_center_id, bt.factory_id) shipmentToid,
       coalesce(f.factory_name,bc.center_name) shipmentToname,
       count(bcs.bale_id) nosofBale,
       sum(vbcd.bale_weight)  baleWeightWithSacks,
       (sum(vbcd.sacks_weight) + sum(vbcd.extra_deduction_weight)) tareWeight,
       sum(vbcd.bale_weight) - (sum(vbcd.sacks_weight) + sum(vbcd.extra_deduction_weight)) netWeight

from (select distinct bale_id from buying_center_stock where in_out_status in ('IN', 'OUT_TRANSFER')) bcs
         join (select btim.bale_id,
                      btim.actual_weight,
                      btim.net_weight,
                      btif.from_buying_center_id,
                      bc.center_name,
                      btif.to_buying_center_id,
                      btif.factory_id,
                      btif.chalan_no,
                      btif.chalan_date,
                      btif.transport_id,
                      btif.transport_vendor_id,
                      btim.bale_deliver_status
               from bale_transfer_info btif
                        join bale_transfer_item btim on btif.id = btim.transfer_id
                    and btim.bale_deliver_status='BALE_RECEIVE'
                   and chalan_date between to_date(:fromDate,'yyyy-mm-dd') and to_date(:toDate,'yyyy-mm-dd')
                   and (case when :shipmentId is not null then btif.chalan_no=:shipmentId else true end)
               join buying_center bc on btif.from_buying_center_id = bc.id
                   and (case when :shipmentFrom is not null then bc.id=:shipmentFrom else true end)
                   and (
                       (case
                             when :shipmentTo is not null
                                 then (btif.to_buying_center_id) = :shipmentTo
                             else true end)
                  or  (case
                   when :shipmentTo is not null
                   then (btif.factory_id) = :shipmentTo
                   else true end) )
) bt on bcs.bale_id = bt.bale_id
         join transport t on bt.transport_id = t.id
    and (case when :truckId is not null then t.Id=:truckId else true end)
         join transport_vendor tv on bt.transport_vendor_id = tv.vendor_id
    and (case when :vendorId is not null then tv.vendor_id=:vendorId else true end)
         join v_bale_classification_details vbcd on bcs.bale_id = vbcd.bale_id
         join (select * from grade where grade_provider = 'CLASSIFIER' and season_id =:seasonId
                                     and (:areaId is null or area_id=:areaId)) g on vbcd.grade_id = g.id
         left join buying_center bc on bt.to_buying_center_id = bc.id
         left join depot d on bc.depot_id = d.id
         left join factory f on bt.factory_id = f.id
group by bt.chalan_date, bt.from_buying_center_id, bt.center_name, bt.to_buying_center_id, bt.factory_id, f.factory_name,
         bc.center_name, bt.chalan_no, tv.vendor_id, tv.trade_license_name, t.id, t.vehicle_name, t.vehicle_reg_no
order by bt.chalan_date, bt.chalan_no,t.id
